select a.*, c.Type, c.Exchange, c.Name from
(select Symbol, ROUND(AVG([low]*[volume])/1000000,0) Turnover, AVG(TradeCount)/1000 TradeCountThs
from DayPolygon 
where year(date) between 2020 and 2025
and [close]>=5 and [low]*[volume]>=50000000 and TradeCount>=10000
group by symbol
having count(*)=1508) a
left join splits b on a.Symbol=b.Symbol and year(b.date) between 2020 and 2025
inner join SymbolsPolygon c on a.Symbol=c.Symbol and c.[To] is null and c.IsTest is null
where b.Symbol is null
